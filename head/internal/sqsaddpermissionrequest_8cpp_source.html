150 /**
151  * @brief Set a permissions to be added for a given principal.
152  *
153  * Note, this adds or removes the \a action permission for \a accountId, within
154  * the context of this request only. That is, if the principal \a accountId
155  * already has the \a action permission granted on the remote SQS service, this
156  * request will not be modifying that existing permission; see
157  * SqsRemovePermissionRequest for that.
158  *
159  * @param accountId Principal to set permssions for.
160  * @param action Action to permit or not.
161  * @param permitted Whether \a action should be permitted or not for \a accountId.
162  */
163 void SqsAddPermissionRequest::setPermission(
164  const QString &accountId, const PermissibleAction action, const bool permitted)
165 {
166  PermissibleActions actions = permissions(accountId);
167  if (permitted) {
168  actions |= action;
169  } else {
170  actions &= ~action;
171  }
172  setPermissions(accountId, actions);
173 }

264 QUrlQuery SqsAddPermissionRequestPrivate::urlQuery() const
265 {
266  QUrlQuery query = SqsRequestPrivate::urlQuery();
267  int index = 1; // SQS attributes are 1-indexed.
268  for (SqsAddPermissionRequest::PermissionsMap::const_iterator iter = permissions.cbegin();
269  iter != permissions.cend(); ++iter)
270  {
271  SqsAddPermissionRequest::PermissibleActions actions = iter.value();
272  if (!actions) continue;
273 
274  // Add the specified permissions.
275  #define IF_ACTION_SET_PARAMETER(action) \
276  if (actions.testFlag(SqsAddPermissionRequest::action##Action)) { \
277  while ((query.hasQueryItem(QString(AWS_ACCOUNT_ID_MEMBER_N).arg(index)))|| \
278  (query.hasQueryItem(QString( ACTION_NAME_MEMBER_N).arg(index)))) { \
279  ++index; \
280  } \
281  query.addQueryItem( \
282  QString(AWS_ACCOUNT_ID_MEMBER_N).arg(index), iter.key()); \
283  query.addQueryItem( \
284  QString(ACTION_NAME_MEMBER_N).arg(index), \
285  toString(static_cast<SqsRequest::Action>(SqsAddPermissionRequest::action##Action))); \
286  actions &= ~SqsAddPermissionRequest::action##Action; \
287  ++index; \
288  }
289  IF_ACTION_SET_PARAMETER(SendMessage)
290  IF_ACTION_SET_PARAMETER(ReceiveMessage)
291  IF_ACTION_SET_PARAMETER(DeleteMessage)
292  IF_ACTION_SET_PARAMETER(ChangeMessageVisibility)
293  IF_ACTION_SET_PARAMETER(GetQueueAttributes)
294  IF_ACTION_SET_PARAMETER(GetQueueUrl)
295  #undef IF_ACTION_SET_PARAMETER
296  Q_ASSERT(actions); // Sanity check.
297  }
298  return query;
299 }
