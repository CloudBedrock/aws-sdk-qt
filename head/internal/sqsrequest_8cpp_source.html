243 /**
244  * @brief Build a network request object for this SQS request.
245  *
246  * This SQS implementation builds request URLs by combining the common query
247  * parameters (such as Action and Version), with any that have been added (via
248  * setParameter) by child classes.
249  *
250  * @param endpoint AWS endpoint to build this request for.
251  *
252  * @return A network request for this SQS request using the given \a endpoint.
253  */
254 QNetworkRequest SqsRequest::unsignedRequest(const QUrl &endpoint) const
255 {
256  Q_D(const SqsRequest);
257  QUrlQuery query;
258  query.addQueryItem(QLatin1String("Action"), actionString());
259  query.addQueryItem(QLatin1String("Version"), apiVersion());
260  for (QVariantMap::const_iterator iter = d->parameters.cbegin(); iter != d->parameters.cend(); ++iter) {
261  const QString parameterName = iter.key();
262  const QVariant parameterValue = iter.value();
263  if (parameterValue.type() == QVariant::Map) {
264  const QVariantMap map = parameterValue.toMap();
265  int index = 1;
266  for (QVariantMap::const_iterator iter = map.cbegin(); iter != map.cend(); ++iter, ++index) {
267  query.addQueryItem(parameterName.arg(index).arg(QLatin1String("Name")), iter.key());
268  query.addQueryItem(parameterName.arg(index).arg(QLatin1String("Value")), iter.value().toString());
269  }
270  } else {
271  query.addQueryItem(parameterName, parameterValue.toString());
272  }
273  }
274 
275  QUrl url(endpoint);
276  url.setQuery(query);
277  return QNetworkRequest(url);
278 }
