184 /**
185  * @brief Build a network request object for this SQS request.
186  *
187  * This SQS implementation builds request URLs by combining the common query
188  * parameters (such as Action and Version), with any that have been added (via
189  * setParameter) by child classes.
190  *
191  * @param endpoint AWS endpoint to build this request for.
192  *
193  * @return A network request for this SQS request using the given \a endpoint.
194  */
195 QNetworkRequest SqsRequest::unsignedRequest(const QUrl &endpoint) const
196 {
197  Q_D(const SqsRequest);
198  QUrlQuery query;
199  query.addQueryItem(QLatin1String("Action"), actionString());
200  query.addQueryItem(QLatin1String("Version"), apiVersion());
201  for (QVariantMap::const_iterator iter = d->parameters.cbegin(); iter != d->parameters.cend(); ++iter) {
202  const QString parameterName = iter.key();
203  const QVariant parameterValue = iter.value();
204  if (parameterValue.type() == QVariant::Map) {
205  const QVariantMap map = parameterValue.toMap();
206  int index = 1;
207  for (QVariantMap::const_iterator iter = map.cbegin(); iter != map.cend(); ++iter, ++index) {
208  query.addQueryItem(parameterName.arg(index).arg(QLatin1String("Name")), iter.key());
209  query.addQueryItem(parameterName.arg(index).arg(QLatin1String("Value")), iter.value().toString());
210  }
211  } else {
212  query.addQueryItem(parameterName, parameterValue.toString());
213  }
214  }
215 
216  QUrl url(endpoint);
217  url.setQuery(query);
218  return QNetworkRequest(url);
219 }
