150 /**
151  * This function parses XML elements like:
152  *
153  * @code{xml}
154  * <Error>
155  * <Type>Sender</Type>
156  * <Code>AccessDenied</Code>
157  * <Message>Access to the resource http://sqs.us-east-1.amazonaws.com/ is denied.</Message>
158  * <Detail/>
159  * </Error>
160  * @endcode
161  *
162  * @note Unlike most other SQS responses, the SQS ErrorResponse does not wrap
163  * the RequestId element in a ResponseMetadata element.
164  *
165  * @see http://queue.amazonaws.com/doc/2012-11-05/QueueService.wsdl
166  */
167 void SqsErrorPrivate::parse(QXmlStreamReader &xml)
168 {
169  if ((xml.isStartDocument()) && (!xml.readNextStartElement())) {
170  return;
171  }
172  Q_ASSERT(xml.name() == QLatin1String("Error"));
173  while (xml.readNextStartElement()) {
174  if (xml.name() == QLatin1String("Type")) {
175  type = typeFromString(rawType = xml.readElementText());
176  } else if (xml.name() == QLatin1String("Code")) {
177  code = codeFromString(rawCode = xml.readElementText());
178  } else if (xml.name() == QLatin1String("Message")) {
179  message = xml.readElementText();
180  } else if (xml.name() == QLatin1String("Detail")) {
181  detail = AwsAbstractResponse::toVariant(xml);
182  } else {
183  qWarning() << "ignoring" << xml.name();
184  xml.skipCurrentElement();
185  }
186  }
187 }

189 SqsError::ErrorCode SqsErrorPrivate::codeFromString(const QString &code)
190 {
191  #define IfStringReturnErrorResponse(str) \
192  if (code == QLatin1String(#str)) return SqsError::str
193  IfStringReturnErrorResponse(AccessDenied);
194  IfStringReturnErrorResponse(IncompleteSignature);
195  IfStringReturnErrorResponse(InternalFailure);
196  IfStringReturnErrorResponse(InvalidAction);
197  IfStringReturnErrorResponse(InvalidClientTokenId);
198  IfStringReturnErrorResponse(InvalidParameterCombination);
199  IfStringReturnErrorResponse(InvalidParameterValue);
200  IfStringReturnErrorResponse(InvalidQueryParameter);
201  IfStringReturnErrorResponse(MalformedQueryString);
202  IfStringReturnErrorResponse(MissingAction);
203  IfStringReturnErrorResponse(MissingAuthenticationToken);
204  IfStringReturnErrorResponse(MissingParameter);
205  IfStringReturnErrorResponse(OptInRequired);
206  IfStringReturnErrorResponse(RequestExpired);
207  IfStringReturnErrorResponse(ServiceUnavailable);
208  IfStringReturnErrorResponse(Throttling);
209  IfStringReturnErrorResponse(ValidationError);
210  #undef IfStringReturnErrorResponse
