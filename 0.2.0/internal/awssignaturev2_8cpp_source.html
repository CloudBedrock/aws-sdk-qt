52 void AwsSignatureV2::sign(const AwsAbstractCredentials &credentials, const QNetworkAccessManager::Operation operation,
53  QNetworkRequest &request, const QByteArray &data) const
54 {
55  Q_UNUSED(data) // Not included in V2 signatures.
56  Q_D(const AwsSignatureV2);
57 
58  // Set the AccessKeyId, SignatureMethod, SignatureVersion and Timestamp query items, if not already.
59  d->adornRequest(request, credentials);
60 
61  // Calculate the signature.
62  const QByteArray stringToSign = d->canonicalRequest(operation, request.url());
63  const QString signature = QString::fromUtf8(QUrl::toPercentEncoding(QString::fromUtf8(
64  QMessageAuthenticationCode::hash(stringToSign, credentials.secretKey().toUtf8(),
65  d->hashAlgorithm).toBase64())));
66 
67  // Append the signature to the request.
68  QUrl url = request.url();
69  url.setQuery(url.query() + QLatin1String("&Signature=") + signature);
70  request.setUrl(url);
71 }

106 /**
107  * @internal
108  *
109  * @brief Add AWS Signature Version 2 adornments to an AWS request.
110  *
111  * In addition to service-specific request parameters, Amazon requires that version
112  * 2 signatures contain a number of common query parameters. This functions adds
113  * those query parameters to \a request if they're not already present.
114  *
115  * The query parameters added by this function, as required by Amazon, are:
116  * * `AWSAccessKeyId` - set to \a credentials.accessKeyId().
117  * * `SignatureMethod` - set to `HMAC-SHA1` or `HMAC-SHA256`.
118  * * `SignatureVersion` - set to `2`.
119  * * `Timestamp` - set to a current UTC timestamp in an ISO 8601 format, like
120  * `2013-10-30T12:34:56Z`.
121  *
122  * @param request Request to adorn.
123  * @param credentials Credentials to use when adorning \a request.
124  *
125  * @see signatureMethod
126  * @see http://docs.aws.amazon.com/general/latest/gr/signature-version-2.html
127  */
128 void AwsSignatureV2Private::adornRequest(QNetworkRequest &request,
129  const AwsAbstractCredentials &credentials) const
130 {
131  // Set / add the necessary query items.
132  QUrl url = request.url();
133  QUrlQuery query(url);
134  setQueryItem(query, QLatin1String("AWSAccessKeyId"), credentials.accessKeyId());
135  setQueryItem(query, QLatin1String("SignatureVersion"), QLatin1String("2"));
136  setQueryItem(query, QLatin1String("SignatureMethod"), QString::fromUtf8(signatureMethod(hashAlgorithm)));
137  setQueryItem(query, QLatin1String("Timestamp"),
138  QString::fromUtf8(QUrl::toPercentEncoding(
139  QDateTime::currentDateTimeUtc().toString(QLatin1String("yyyy-MM-ddThh:mm:ssZ"))
140  )),
141  false); // Don't warn if its already set to something else.
142 
143  // If we've touched the query items (likely), then update the request.
144  if (query != QUrlQuery(url)) {
145  qDebug() << url;
146  url.setQuery(query);
147  qDebug() << url;
148  request.setUrl(url);
149  }
150 }

152 /**
153  * @internal
154  *
155  * @brief Create an AWS V2 Signature canonical request.
156  *
157  * This function creates a canonical representation of an AWS request as defined by
158  * Amazon's V2 signature specification.
159  *
160  * For example, for the following HTTP `GET` request:
161  *
162  * https://elasticmapreduce.amazonaws.com?Action=DescribeJobFlows&Version=2009-03-31&AWSAccessKeyId=AKIAIOSFODNN7EXAMPLE&SignatureVersion=2SignatureMethod=HmacSHA256Timestamp=2011-10-03T15%3A19%3A30
163  *
164  * this function will return the following canonical form:
165  *
166  * GET
167  * elasticmapreduce.amazonaws.com
168  * /
169  * AWSAccessKeyId=AKIAIOSFODNN7EXAMPLE&Action=DescribeJobFlows&SignatureMethod=HmacSHA256&SignatureVersion=2&Timestamp=2011-10-03T15%3A19%3A30&Version=2009-03-31
170  *
171  * @note All URL components are encoded to UTF-8, as required by Amazon.
172  *
173  * @param operation The HTTP method being requested.
174  * @param url The URL being request.
175  *
176  * @return An AWS V2 Signature canonical request.
177  *
178  * @see http://docs.aws.amazon.com/general/latest/gr/signature-version-2.html
179  */
180 QByteArray AwsSignatureV2Private::canonicalRequest(const QNetworkAccessManager::Operation operation,
181  const QUrl &url) const
182 {
183  return httpMethod(operation).toUtf8() + '\n' +
184  url.host().toUtf8() + '\n' +
185  canonicalPath(url).toUtf8() + '\n' +
186  canonicalQuery(QUrlQuery(url));
187 }

189 /**
190  * @brief Create an AWS V2 Signature method designation.
191  *
192  * This function returns a signature method designation, as defined by Amazon, for
193  * use with V2 signatures.
194  *
195  * For example, if the algorith is `QCryptographicHash::Sha256`, this function will
196  * return `HmacSHA256`.
197  *
198  * @note Amazon only supports two algorithms for V2 signatures - SHA1 and SHA256.
199  *
200  * @param algorithm The hash algorithm to get the canonical designation for.
201  *
202  * @return An AWS V2 Signature method designation.
203  *
204  * @see http://docs.aws.amazon.com/general/latest/gr/signature-version-2.html
205  */
206 QByteArray AwsSignatureV2Private::signatureMethod(const QCryptographicHash::Algorithm algorithm) const
207 {
208  switch (algorithm) {
209  case QCryptographicHash::Sha1: return "HmacSHA1";
210  case QCryptographicHash::Sha256: return "HmacSHA256";
211  default:
212  Q_ASSERT_X(false, Q_FUNC_INFO, "invalid algorithm");
213  return "invalid-algorithm";
214  }
215 }
